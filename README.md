# ros_course
ros course projects

1. create workspace
   - $mkdir "workspace_name"
2. clone repo into <workspace> src
   - cd "workspace_name"
   - $git clone https://github.com/AndreyMinin/ros_course.git src
3. init ros ws (needed only once)
   - $catkin_init_workspace src
4. build workspace
   - $catkin_make
